package com.linkedin.pinot.core.query.utils;

import java.io.Serializable;


public class Pair<T1 extends Serializable, T2 extends Serializable> implements Serializable {
  private static final long serialVersionUID = -2776898111501466320L;

  private T1 _first;
  private T2 _second;

  public Pair(T1 first, T2 second) {
    _first = first;
    _second = second;
  }

  public T1 getFirst() {
    return _first;
  }

  public T2 getSecond() {
    return _second;
  }

  public void setFirst(T1 first) {
    _first = first;
  }

  public void setSecond(T2 second) {
    _second = second;
  }

  @Override
  public String toString() {
    return "first=" + _first + ", second=" + _second;
  }

}
